-- Geração de Modelo físico
-- Sql ANSI 2003 - brModelo.

CREATE DATABASE Webfono

CREATE TABLE Venda (
id_venda int PRIMARY KEY,
data timestamp,
total_venda float,
id_produto int,
id_paciente int
)

CREATE TABLE Produto (
id_produto int PRIMARY KEY,
nome_produto Varchar(60),
descricao Varchar(150),
quantidade int,
preco float,
situacao Varchar(30)
)

CREATE TABLE Paciente (
id_paciente int PRIMARY KEY,
nome_paciente Varchar(60),
cpf Varchar(16),
RG Varchar(16),
gmail Varchar(70),
nascimento date,
telefone char(14),
endereco Varchar(80),
bairro Varchar(30),
cidade Varchar(40),
cep Char(9)
)

CREATE TABLE Atendimento (
id_atendimento int PRIMARY KEY,
data date,
descricao Varchar(150),
hora time,
id_paciente int
)

CREATE TABLE login (
id_login int PRIMARY KEY,
usuario Varchar(50),
email varchar(90),
senha Varchar(20)
)

ALTER TABLE Venda ADD FOREIGN KEY(id_produto) REFERENCES Venda (id_venda)
ALTER TABLE Venda ADD FOREIGN KEY(id_paciente) REFERENCES Venda (id_venda)
ALTER TABLE Atendimento ADD FOREIGN KEY(id_paciente) REFERENCES Atendimento (id_atendimento)
